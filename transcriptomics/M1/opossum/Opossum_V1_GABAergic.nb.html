  arrange(desc(group)) %>%
  mutate(prop = value / sum(data$value) *100) %>%
  mutate(ypos = cumsum(prop)- 0.5*prop )

  geom_bar(stat="identity", width=1) +
  coord_polar("y", start=0) +
  theme_void() + 
  geom_text(aes(y = ypos, label = sprintf("%.1f%%", prop)), color = "white", size=5) +
